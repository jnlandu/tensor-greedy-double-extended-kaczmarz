\section{Abstract}
\begin{frame}{Abstract}
Tensors are intensively used to represent images, videos, and any higher dimensional object. In image reconstruction, given a blurred and noisy images, the reconstruction of the image can be expressed as a tensor linear systems. In this note, we present a tensor greedy double extended Kaczmarz (TGDBEK) for solving large scale tensor linear systems. Unlike the block-based methods TREBK, TREGBK, TREABK encountered in the literature which more or less require a pred-defined partition of the tensor system, the TGDBEK \textbf{does not require any pre-defined partitions of the tensor systems}. It employs \textbf{the greedy principle on the tensor systems to dynamically construct a relatively small subset of rows and columns slices}. Then updates the iterates  using  a REK update style.  We provide the convergence guarantees of the proposed method. We have sown that the proposed method converges exponentially in expectation to the minimum-norm solutions. Numerical examples are conducted to asses the effectiveness of the proposed method.
\end{frame}

\subsection{Tensors \& 3D Tensors}
\begin{frame}{Tensors \& 3D Tensors}
We follow the notations and take the concepts from Kilmer et al. \cite{kilmer2013third}.
\begin{itemize}
    \item A tensor is a natural generalization of a matrix in higher dimension. See fig. \ref{fig:3d-tensors}.
    \item We borough our notations from \cite{kilmer2013third, du2021rrek}. We denote tensors by calligraphic letters $\fm A$.
    \item We only  consider third-order tensors, i.e. $\fm A \in \r^{N_1\times N_2\times N_3}$
    \item  Given a tensor $\fm A$, let $\m A_i:=\fm A_{:, :, i}$. The block circulant matrix of $\fm A$ is defined as \cite{kilmer2013third}:
$$
    \bcirc(\mcala):=\bem\mbf A_1 &\mbf A_{N_3} & \cdots & \mbf A_2\\ \mbf A_2 & \mbf A_1 & \cdots &\mbf A_3\\ \vdots &\vdots & \ddots &\vdots\\ \mbf A_{N_3}& \mbf A_{N_3-1}& \cdots &\mbf A_1\eem\in\mbbr^{N_1N_3\times N_2N_3}.

\end{itemize}

\subsection{T-product}
\begin{frame}{T-product}
\begin{definition}[1, \cite{kilmer2013third}]
    For $\mcala\in\mbbr^{N_1\times N_2\times N_3}$ and $\mcalb\in\mbbr^{N_2\times K\times N_3}$, the {\it \textbf{t-product}} between $\mcala$ and $\mcalb $ is  a $ N_1\times K\times N_3$ tensor  defined and denoted  $\mcala*\mcalb$ 
$$
\mcala*\mcalb:=\fold(\bcirc(\mcala)\unfold(\mcalb))\in \r^{N_1\times K\times N_3}
$$
\end{definition}
\begin{itemize}
    \item The product $\bcirc(\mcala) \unfold(\mcalB)$ is a usual matrix product, which results to a  block matrix that we fold to get a 3d tensor.
    % \item The resulting matrix is a block matrix which then is folded to a tensor.
\end{itemize}
\end{frame}

\subsection{Some definitions and properties.}
\begin{frame}{Some definitions and properties.}
We summarize below some definitions and properties that we will use in this note.
\begin{definition}[\cite{kilmer2013third, du2021rrek, an2025randomized}]
Let $\mcala \in \r^{N_1\times \times N_2 \times N_3}$ and $\mcalb \in \r^{N_2\times K\times N_3}$. Then the t-product between $\fm A $ and $\fm B$ satisfy the following properties
\begin{enumerate}
    \item The separability in the first dimension, that is 
    \begin{align*}
        \left( \fm A * \fm B\right)_{i, :, :} = \fm A_{i, :, :} * \fm B.
    \end{align*}
    \item  The component form decomposition on the second dimension
    \begin{align}
        \fm A * \fm B = \sum_{j=1}^{N_2} \fm A_{:, j, :} * \fm B_{j, :, :}.
    \end{align}
\end{enumerate}
\end{definition}
\end{frame}

\begin{frame}{}
\begin{definition}[transpose]
The transpose of $\mcala\in\mbbr^{N_1\times N_2\times N_3}$, denoted by $\mcala^\top$, is the $N_2\times N_1\times N_3$ tensor obtained by transposing each of the frontal slices and then reversing the order of transposed frontal slices $2$ through $N_3$.
For $\mcala\in\mbbr^{N_1\times N_2\times N_3}$, we have  \beq\label{ttop}\bcirc(\mcala^\top)=(\bcirc(\mcala))^\top.\eeq %and \beq\label{eqj}\mcala*\mcala^\top=\sum_{j=1}^{N_2}\mcala_{:,j,:}*(\mcala_{:,j,:})^\top.\eeq 
\end{definition}
\begin{definition}[inner product]
The inner product between $\mcala$ and $\mcalb$ in $\mbbr^{N_1\times N_2\times N_3}$ is defined as $$\la\mcala,\mcalb\ra:=\sum_{i,j,k}\mcala_{ijk}\mcalb_{ijk}.$$ 
\end{definition}
\end{frame}

\begin{frame}{}
\begin{definition}  For $\fm A \in \r^{N_1 \times N_2 \times N_3}$, its spectral norm and the Frobenius norm are defined as follows
\begin{align}
    \Vert \fm A \Vert_2 &  := \Vert \bcirc (\fm A ) \Vert_2\\
    \Vert \fm A \Vert_F &:= \sqrt{\langle \fm A , \fm A \rangle} = \sqrt{\sum_{i,j,k}}\fm A(i,j,k)^2
\end{align}
\end{definition} 
\begin{lemma}[\cite{kilmer2013third}]
    For $\fm A\in \r^{N_1\times N_2 \times N_3}$ and $\fm B\in \r^{N_1\times N_2 \times N_3}$, it holds that 
    \begin{align}
        \Vert \fm A * \fm B \Vert_F\leq \Vert \fm A \Vert_2 \Vert \fm B \Vert_F
    \end{align}
\end{lemma}
\end{frame}
\begin{frame}{}
\begin{definition} For $\fm A\in \r^{N_1\times N_2\times N_3}$, its pseudoinverse, denoted by $\fm A^\dagger$, is the $N_2\times N_1\times N_3$ tensor satisfying
\begin{align}
    \bcirc(\fm A^\dagger) = \bcirc(\fm A )^\dagger
\end{align}
\end{definition}
\begin{lemma} For $\fm A\in \r^{N_1\times N_2\times N_3}$ and  $\fm B\in \r^{N_2\times K\times N_3}$, the following holds
\begin{align}
\fm A^\top * \fm A * \fm A^\dagger* \fm B = \fm A^\top * \fm B
\end{align}
\end{lemma}
In fact, we know that  $\fm A * \fm A^\top $ is symmetric, $\fm A * \fm A^\dagger * \fm A = \fm A$ and $(\fm A^\dagger)^\top = (\fm A^\top)^\dagger $. We have
\begin{align*}
    \fm A^\top * \fm A * \fm A^\dagger * \fm B  = \fm A^\top *\fm (A^\dagger)^\top * \fm A^\top  * \fm B = \fm A^\top * (\fm A^\top)^\dagger * \fm A = \fm A^\top * \fm B
\end{align*}
\end{frame}
\begin{frame}{}
\begin{lemma}
For $\fm A\in \r^{N_1\times N_2\times N_3}$ and $\fm X\in $ range($\fm A$),  the following holds 
\begin{align}
    \Vert \fm A^\top * \fm X \Vert_F^2 &  \geq \sigma_{\min}^2(\bcirc(\fm A))\Vert \fm X \Vert_F^2
\end{align}
With the algebra defined above, it becomes clear that the t-product is for tensors what the product (composition) os for matrices.
\end{lemma}
\end{frame}

\subsection{Problem \& Literature }
\begin{frame}{Problem \& Literature }
\begin{itemize}
    \item We aim to solve the linear system of the form
    \begin{align}
        \label{eq:eq-1}
            \fm A * \fm X =  \fm B, \; \fm B = \overline{\fm B} + \varepsilon
\end{align}
where $\varepsilon$ is a noisy, and $\overline{\fm B}$ is the clean right-hand side. 
\item This problem arises in image reconstruction \cite{kilmer2013third, du2021rrek,an2025randomized}.
\item To solve this problem, Guang-Xin Huang and collaborators in \cite{huang2024tensor} proposed block-based tensor randomized extended  Kaczmarz methods, namely TREBK, TREGBK.
\item Unlike the TREBK, the TREGBK uses a greedy strategy which \textbf{discriminates blocks}  and  projects the current iterate  on a set of block whose Frobenius norm is bigger than some threshold.
\end{itemize}
\end{frame}
\begin{frame}{}
\begin{itemize}
    \item Subsequently, Liyuan Ann et al. \cite{an2025randomized} adapted recently the randomized extended average block Kaczmarz (REABK) by  Du et al. \cite{du2020randomized} to the tensor framework under the t-product to obtain the tensor randomized extended average Kaczmarz (TREABK).
    \item Unlike the TREGBK or TREBK, the TREABK doesn't project the current iterates on the block hyperplanes forming the selected block. But, it uses a weighted convex combination of projection the the current iterate on each selected hyperplane. 
\end{itemize}
\end{frame}

\begin{frame}{}
\begin{itemize}
    \item The averaging idea intends to ameliorate the convergence of the TREABK compare to TREBK and TREGBK, as was the authors in \cite{an2025randomized} observed.
    \item What these methods have in common is the fact the require a  \textbf{predefined partition of the tensor systems}.
    \item The partition choice can affect the performance of the methods, as blocks can differ one to an other.
    \item the TREGBK introduces greedy block sampling. One of the advantages of using a greedy technique is it specifies a discrimination criterion on (blocks of) hyperplanes, inform the method to only project on the "best" of those. It likely results to less running time and less iteration. Hence, a better convergence decay \cite{bai2018greedy, bai2021greedy} compared to non-greedy methods.
\end{itemize}
\end{frame}


\section{Proposed method: Tensor greedy double  block extended Kaczmarz}
\subsection{TGDBEK algorithm}
\begin{frame}[t]{TGDBEK algorithm}
\scriptsize
\setlength{\tabcolsep}{2pt}
\renewcommand{\arraystretch}{1.05}

\begin{center}
\begin{tabular*}{\linewidth}{@{\extracolsep{\fill}}l@{}}
\toprule
{\bf Algorithm 4:} Tensor Greedy Double Block Extended Kaczmarz (TGDBEK)\\
\midrule
{\bf Input:} $\mathcal{A}\!\in\!\mathbb{R}^{N_1\times N_2\times N_3}$, $\mathcal{B}\!\in\!\mathbb{R}^{N_1\times K\times N_3}$, iters $\ell$, $\eta\in(0,1]$.\\
{\bf Init:} $\mathcal{X}^{(0)}=\mathcal{O}$, $\mathcal{Z}^{(0)}=\mathcal{B}$.\\
{\bf for} $k=0,1,\ldots,\ell-1$ {\bf do}\\
\quad $\displaystyle \varepsilon_k^{z}=\eta\max_{j\in[N_2]}
\frac{\|(\mathcal{A}_{:,j,:})^\top * \mathcal{Z}^{(k)}\|_{F}^2}{\|\mathcal{A}_{:,j,:}\|_{F}^2}$\\[-0.2em]
\quad $\displaystyle U_k=\Big\{j\in[N_2]:\|(\mathcal{A}_{:,j,:})^\top * \mathcal{Z}^{(k)}\|_{F}^2\ge
\varepsilon_k^{z}\|\mathcal{A}_{:,j,:}\|_{F}^2\Big\}$\\[-0.2em]
\quad $\displaystyle \mathcal{Z}^{(k+1)}=\mathcal{Z}^{(k)}-\mathcal{A}_{:,U_k,:}*(\mathcal{A}_{:,U_k,:})^\dagger*\mathcal{Z}^{(k)}$\\[-0.1em]
\quad $\displaystyle \varepsilon_k^{x}=\eta\max_{i\in[N_1]}
\frac{\|\mathcal{B}_{i,:,:}-\mathcal{Z}^{(k+1)}_{i,:,:}-\mathcal{A}_{i,:,:}*\mathcal{X}^{(k)}\|_{F}^2}{\|\mathcal{A}_{i,:,:}\|_{F}^2}$\\[-0.2em]

\subsection{Convergence theorem}
\begin{frame}{Convergence theorem}

\begin{theorem} Assume the system  \eqref{eq:eq-1} is inconsistent and let $\fm B^\perp := \fm B - \fm A \fm A^\dagger \fm B$. The tensor sequence $\fm X^{(k)}$ generated  by TGDBEK algorithm  converges to the  tensor least-square solution $\fm X_\ast = \fm A^\dagger * \fm B$. Moreover, for $k\geq 1$, the following holds
\begin{align}
    \Vert \fm X ^{(k+1)} - \fm X_\ast \Vert_F^2 \leq \alpha^{k+1}\Vert \fm X_0 - \fm X_\ast \Vert_F^2 + \frac{\alpha^{k+1}-\beta^{k+1}}{\alpha - \beta}\Vert \fm B^\perp \Vert_F^2
\end{align}
where $ \alpha = 1 - \theta \frac{\sigma_{\min}^2(\bcirc(\fm A^\top))}{\sigma^2_{\max}(\bcirc(\fm A^\top ))}$, $\beta = 1- \frac{\sigma^2_{\max}(\fm A)}{\Vert \fm A \Vert^2_F - \Vert \fm A(\overline{U}) \Vert_F^2}$, $\gamma = \sigma^2_{\min}(\fm A^{\fm J}), \; \theta \in (0,1]$ and $\eta \in (0, 1]$.
\end{theorem} 
\end{frame}
\begin{frame}{}
    \begin{proof} From Lemma 5, we obtain 
\begin{align*}
    \fm A^\top *\fm B^\perp  =\fm A^\top  *\fm B - \fm A^\top * \fm A * \fm A^\dagger * \fm B =\fm A ^\top * \fm B  - \fm A ^\top * \fm B = 0
\end{align*}
In particular, for any $U_k \subset [N_2]$  we get $\fm A(:, U_k, :)^\top \fm B^\perp = 0$.
Now, using the fact $\fm A * \fm A^\dagger = \left(\fm A * \fm  A^\dagger \right)^\top = (\fm A^\dagger)^\top * \fm A ^\top = (\fm A^\top )^\dagger * \fm A^\top   $  and the $z$-update in Alg 1, we can get
\begin{align*}
    \fm Z^{(k+1)} - \fm B^\perp  & = \left(\fm Z^{(k)} - \fm B^\perp\right) - \left(\fm A^\top_{:, {U_k}, :} \right)^\dagger* \fm A^\top_{:, U_k, :} *\left( \fm Z ^{(k)}-\fm B^\perp \right)\\
    & = \left(\fm I_{:, U_k, :} - \left( \a{U_k}^\top\right) ^\dagger \a{U_k}^\top\right)*\left( \fm Z^{(k)}- \fm B^\perp \right).
\end{align*}
Since $\left( \a{U_k}^\top\right) ^\dagger \a{U_k}^\top$  is a $t-$ tensor projector,  using the Pythagorean theorem,  we can obtain

\end{proof}
\end{frame}

\begin{frame}{}
\begin{align}
    \Vert   \fm Z^{(k+1)} - \fm B^\perp  \Vert_F^2  = \Vert  \fm Z^{(k)} - \fm B^\perp  \Vert_F^2 - \Vert  \left(\fm A^\top_{:, {U_k}, :} \right)^\dagger* \fm A^\top_{:, U_k, :} *\left( \fm Z ^{(k)}-\fm B^\perp \right) \Vert_F^2. \label{eq:eq-1b}
\end{align}
Using Lemma (), we  note that 
\begin{align}
    \Vert \left(\a{U_k}^\top \right)^\dagger * \a{U_k}^\top *\left(\fm Z^{(k)}-\fm B^\perp \right) \Vert_F^2 & =  \Vert \left(\a{U_k}^\top \right)^\dagger * \a{U_k}^\top *\fm Z^{(k)}\Vert_F^2 \notag\\
    & \geq \sigma_{\min}^2((\a{U_k}^\top) ^\dagger)\Vert \a{U_k}^\top * \fm Z^{(k)} \Vert_F^2 \notag\\
    & = \sigma_{\min}^2(\bcirc(\a{U_k}^\top) ^\dagger) \sum_{j\in U_k} \Vert \a j ^\top * \fm Z^{(k)} \Vert_F^2 \notag\\
    & \geq \sigma_{\min}^2(\bcirc(\a{U_k}^\top) ^\dagger) \epsilon^z_k \Vert \a{U_k} \Vert_F^2 \notag\\
    & =  \frac{\Vert \a{U_k} \Vert_F^2}{\sigma_{\max}^2(\bcirc(\a{U_k}^\top))} \epsilon_k^z .\label{eq:eq-2}
\end{align}
Now,  using the  lemma (), we observe that
\begin{align*}
    \a{U_{k-1}}^\top \fm Z^{(k)} & = \a{U_{k-1}}^\top \left( \fm Z^{(k-1)} - \a{U_{k-1}}*\a{U_{k-1}}^\dagger \fm Z^{k-1} \right)\\
    & =  \a{U_{k-1}}^\top*\fm Z^{(k-1)} - \a{U_{k-1}}^\top *\a{U_{k-1}}\a{U_{k-1}}^\dagger* \fm Z^{k-1} = \fm O, 
\end{align*}
% so that for all $k=0, 1, \ldots,$, we can get
\end{frame}
\begin{frame}{}
so that for all $k=0, 1, \ldots,$, we can get
\begin{align*}
    \Vert \fm A^\top \fm Z^{(k)}\Vert_F^2  & = \sum_{j \in U_k} \Vert \a j ^\top  * \fm Z^{(k)} \Vert_F^2 + \sum_{j\in U^c_{k-1}}\Vert \a j ^\top  * \fm Z^{(k)} \Vert_F^2 \\
    &=  \sum_{j\in U^c_{k-1}}\Vert \a j ^\top  * \fm Z^{(k)} \Vert_F^2  =  \sum_{j\in U^c_{k-1}} \frac{ \Vert \a j ^\top  * \fm Z^{(k)} \Vert_F^2}{ \Vert \a j \Vert_F^2}\Vert \a j \Vert_F^2\\
    & \leq \max_{j \in [N_2]}\left\{  \frac{ \Vert \a j ^\top  * \fm Z^{(k)} \Vert_F^2}{ \Vert \a j \Vert_F^2} \right\}\left(\Vert \fm A \Vert_F^2 - \Vert \a{U_k} \Vert_F^2 \right).
\end{align*}
Hence
\begin{align}
    \epsilon^z_k = \eta \max_{j \in [N_2]}\left\{  \frac{ \Vert \a j ^\top  * \fm Z^{(k)} \Vert_F^2}{ \Vert \a j \Vert_F^2} \right\} &\geq \eta \frac{ \Vert \fm A^\top \fm Z^{(k)}\Vert_F^2}{\Vert \fm A \Vert_F^2 - \Vert \a{U_k} \Vert_F^2} \notag\\
    & \geq \eta\frac{ \sigma_{\min}^2(\bcirc(\fm A^\top)) \Vert \fm Z^{(k)}-\fm B^\perp \Vert_F^2}{\Vert \fm A \Vert_F^2 - \Vert \a{U_k} \Vert_F^2},  \label{eq:eq-3}
\end{align}
where in the last inequality we used $ \Vert \fm A^\top *  \fm Z^{(k)}\Vert_F^2 =  \Vert \fm A^\top * \left( \fm Z^{(k)} - \fm B^\perp \right)\Vert_F^2 \geq \sigma_{\min}^2(\bcirc(\fm A^\top)) \Vert \fm Z^{(k)}-\fm B^\perp \Vert_F^2 $.   
\end{frame}
\begin{frame}{}
 Substitute \eqref{eq:eq-3} in \eqref{eq:eq-2} we can obtain
\begin{align}
     \Vert \left(\a{U_k}^\top \right)^\dagger * \a{U_k}^\top *\left(\fm Z^{(k)}-\fm B^\perp \right) \Vert_F^2 & \geq \frac{\eta \Vert \a{U_k} \Vert_F^2}{\Vert \fm A \Vert_F^2 - \Vert \a{U_k} \Vert_F^2} \frac{\sigma_{\min}^2(\bcirc(\fm A^\top)) \Vert \fm Z^{(k)}-\fm B^\perp \Vert_F^2}{\sigma_{\max}^2(\bcirc(\a{U_k}^\top))}. \label{eq:eq-4}
\end{align}
Let $\Vert \a{\overline{U}} \Vert_F^2 := \max_k \{ \Vert \a{U_k}\Vert_F^2\}$ so that, combining \eqref{eq:eq-1b} and  \eqref{eq:eq-4} above, we can finally obtain the upper bound
\begin{align}
     \Vert   \fm Z^{(k+1)} - \fm B^\perp  \Vert_F^2  & \leq  \left( 1 - \frac{\eta \Vert \a{U_k} \Vert_F^2}{ \sigma_{\max}^2(\bcirc(\a{U_k}^\top)) } \frac{\sigma_{\min}^2(\bcirc(\fm A^\top))}{\Vert \fm A \Vert_F^2 - \Vert \a{\overline{U}} \Vert_F^2}   \right) \Vert  \fm Z^{(k)} - \fm B^\perp \Vert_F^2  \notag \\
     & \leq \left( 1 - \eta  \frac{\sigma_{\min}^2(\bcirc(\fm A^\top))}{\Vert \fm A \Vert_F^2 - \Vert \a{\overline{U}} \Vert_F^2}   \right) \Vert  \fm Z^{(k)} - \fm B^\perp \Vert_F^2 .
\end{align}
where  in the last inequality we used the fact that $\frac{\Vert \a{U_k}\Vert_F^2 }{\sigma_{\max}^2(\bcirc(\a{U_k}))}\geq 1$. The fact that  we can decompose $\b{J_k} = \ai{J_k}*\fm X^\ast + \bp{J_k} $, we can rewrite the the $x-$ update as,
\begin{align*}
   \fm X^{(k+1)} = \x k - \ai{J_k}^\dagger * \left( \a{J_k} *\left(\x k - \fm X^\ast \right) + \left(\z{k+1}{J_k} - \bp{J_k}  \right)\right)
\end{align*}   
\end{frame}
\begin{frame}{}
\scriptsize
Subtracting $\fm X_\ast$  both sides  in the equality above, we obtain
\begin{align}
   \x {k+1} - \fm X_\ast  & =  \x{k} - \fm X_\ast  - \ai{J_k}^\dagger * \left( \a{J_k} *\left(\x k - \fm X^\ast \right) + \left(\z{k+1}{J_k} - \bp{J_k}  \right)\right) \notag\\
   & =  \left( \fm I - \ai{J_k}^\dagger *  \a{J_k} \right) *\left(\x k - \fm X^\ast   \right) +  \ai{J_k}^\dagger* \left(\z{k+1}{J_k} - \bp{J_k}  \right)
\end{align}
Again $\a{J_k}^\dagger \a{J_k}$ is a $t-$ orthogonal projector  (slice-wise  orthogonal in the Fourier  domain), hence the two terms are orthogonal in the Frobenius norm. Therefore,  from the  Pythagorean theorem , we can obtain
\begin{align}
    \Vert \x{k+1} - \fm X_\ast \Vert_F^2 = \Vert \left( \fm I - \ai{J_k}^\dagger *  \a{J_k} \right) *\left(\x k - \fm X^\ast   \right) \Vert_F^2  +  \Vert  \ai{J_k}^\dagger* \left(\z{k+1}{J_k} - \bp{J_k}  \right) \Vert_F^2  \label{eq:eq-5}
\end{align}
Let  $\x{k}-\fm X_\ast = \left( \x{k} - \fm X_\ast \right)_{\text{R}((\a{J_k})^\top)} + \left(\x k - \fm X_\ast \right)_{\text{R}((\a{J_k})^\top)^\perp }$ and $\Vert \left( \x{k} - \fm X_\ast \right)_{\text{R}((\a{J_k})^\top)} \Vert_F^2  = \theta_k \Vert \x k - \fm X_\ast \Vert_F^2$. Then, using the Pythagorean theorem and the  Lemma (), we get :

\Vert \left( \fm I - \ai{J_k}^\dagger *  \a{J_k} \right) *\left(\x k - \fm X^\ast   \right) \Vert_F^2  & = \Vert \x k - \fm X_\ast \Vert_F^2 - \Vert  \ai{J_k}^\dagger *  \a{J_k}  *\left(\x k - \fm X^\ast   \right) \Vert_F^2  \notag\\

& = \left( 1 - \theta_k \frac{\sigma_{\min}^2(\bcirc(\a{J_k}))}{\sigma^2_{\max}(\a{J_k})}\right) \Vert \x k - \fm X_\ast \Vert_F^2 \notag  \\
& \leq \left( 1 - \theta \frac{\sigma_{\min}^2(\bcirc(\a{J}))}{\sigma^2_{\max}(\a{J})}\right) \Vert \x k - \fm X_\ast \Vert_F^2 \label{eq:eq-6}, 
\end{align}
where we denote $J = \{J_0, 1, \ldots, J_k \}$ and $\theta:= \min_{k} \{\theta_k \}$, and $\a{J}$ the stacked or the union block of tensors  $\a{J_k}$. For the second term,  we first note that $\Vert \fm A^\dagger \fm X \Vert_F \leq \Vert \fm A ^\dagger \Vert_2^2 \Vert \fm X \Vert_F^2$.
    
\end{frame}

Then, we obtain
\begin{align}

Then, induction over $k$ yields
\begin{align*}
    \Vert \x{k+1} - \fm X_\ast \Vert_F^2  & \leq \alpha  \Vert \x k - \fm X_\ast \Vert_F^2 + \frac{1}{\gamma}\beta \Vert \fm Z^{(k)} - \fm B^\perp \Vert_F^2\\
    & \leq \alpha \left( \alpha  \Vert \x k - \fm X_\ast \Vert_F^2 + \frac{1}{\gamma}\beta \Vert \fm Z^{(k)} - \fm B^\perp \Vert_F^2 \right) + \beta^2 \frac{1}{\gamma} \Vert \fm Z^{(k-1)} - \fm B^\perp \Vert_F^2\\
    &\vdots\\
    & \leq \alpha^{k+1} \Vert \x 0 - \fm X_\ast \Vert_F^2  + \frac{1}{\gamma}\sum_{l=0}^k \alpha^{l} \beta^{k-l+1}  \Vert \fm Z^{(k-1)} - \fm B^\perp \Vert_F^2\\
    &\leq \alpha^{k+1} \Vert \x 0 - \fm X_\ast \Vert_F^2 + \frac{\alpha^{k+1} -\beta^{k+1}}{\alpha - \beta} \frac{\beta}{\alpha}\Vert \fm B_{R(\fm A)}\Vert_F^2
\end{align*}


\end{frame}
